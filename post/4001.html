$p$ е’Ң $q$ йҖүеҸ–дёҚеҪ“ рҹҳҺ

еҰӮжһң $p$ е’Ң $q$ йҖүеҸ–дёҚеҪ“пјҢдҫӢеҰӮе®ғд»¬д№Ӣй—ҙзҡ„е·®еҖјиҝҮе°ҸпјҢжҲ–иҖ…е®ғд»¬зҡ„жҜ”зү№дҪҚж•°зӣёе·®иҝҮеӨ§пјҢйӮЈд№Ҳж”»еҮ»иҖ…еҸҜд»ҘеҲ©з”ЁиҝҷдәӣејұзӮ№иҝӣиЎҢж”»еҮ»гҖӮ

еҰӮжһң $p$ е’Ң $q$ зҡ„е·®еҖјиҫғе°ҸпјҢеҸҜд»ҘдҪҝз”Ё Fermat еҲҶи§Јжі•еҝ«йҖҹеҲҶи§ЈеҮә $n$пјҢд»ҺиҖҢиҺ·еҸ–еҲ° $p$ е’Ң $q$гҖӮ

Fermat еҲҶи§Јжі•пјҡ
дҫқжҚ®е…¬ејҸ
$$\frac{(p+q)^2}{4} - n = \frac{(p+q)^2}{4} - pq = \frac{(p-q)^2}{4}$$

$|p-q|$ жҜ”иҫғе°Ҹ,йӮЈд№Ҳ $\frac{(p-q)^2}{4}$ д№ҹе°ҸпјҢжүҖд»Ҙ $\frac{(p+q)^2}{4}$ жҺҘиҝ‘дәҺnпјҢеҸҜд»ҘйҖҡиҝҮдёҚж–ӯеўһеҠ  $\frac{(p+q)}{2}$ зҡ„еҖјпјҢзӣҙеҲ° $\frac{(p-q)^2}{4}$ дёәе®Ңе…Ёе№іж–№ж•°гҖӮ

и®ҫ $x = \frac{(p+q)}{2}$, $y = \frac{(p-q)}{2}$
еҪ“ж»Ўи¶і $x^2 - n = y^2$ ж—¶еҚіеҸҜжұӮеҮәж–№зЁӢ

еҰӮжһң $p$ е’Ң $q$ зҡ„жҜ”зү№дҪҚж•°зӣёе·®иҝҮеӨ§пјҢйӮЈд№Ҳж”»еҮ»иҖ…еҸҜд»Ҙе°қиҜ•зӣҙжҺҘеҲҶи§Јиҫғе°Ҹзҡ„зҙ ж•°пјҢиҺ·еҸ–еҲ°е…¶дёӯдёҖдёӘзҙ ж•°еҗҺпјҢеҶҚйҖҡиҝҮйҷӨжі•и®Ўз®—еҮәеҸҰдёҖдёӘзҙ ж•°пјҢд»ҺиҖҢиҺ·еҸ–еҲ° $p$ е’Ң $q$гҖӮ

from sage.all import *
from Crypto.Util.number import *
from itertools import count
from sympy import primerange

# williams's p+1 factorization
def mlucas(v, a, n):
    v1, v2 = v, (v ** 2 - 2) % n
    for bit in bin(a)[3:]:
        if bit == "0":
            v1, v2 = (v1 ** 2 - 2) % n, (v1 * v2 - v) % n
        else:
            v1, v2 = (v1 * v2 - v) % n, (v2 ** 2 - 2) % n
    return v1
 
# зҙ ж•°з”ҹжҲҗеҷЁ
def primegen():
    yield from primerange(2, 10**6)  # з”ҹжҲҗеҲ° 10^6 зҡ„зҙ ж•°пјҢеӨҹз”ЁдәҶ
 
# ж•ҙж•°еҜ№ж•°пјҡilog(x, b) = жңҖеӨ§ж•ҙж•° lпјҢдҪҝеҫ— b^l <= x
def ilog(x, b):
    l = 0
    while x >= b:
        x //= b
        l += 1
    return l
 
# Williams p+1 еҲҶи§Јж”»еҮ»
def Williams_p_plus_1_attack(n):
    for v in count(1):  # дёҚж–ӯе°қиҜ•ж–°зҡ„ v
        for p in primegen():
            e = ilog(isqrt(n), p)
            if e == 0:
                break
            for _ in range(e):
                v = mlucas(v, p, n)
            g = gcd(v - 2, n)
            if 1 < g < n:
                return int(g), int(n // g)
            if g == n:
                break
 
# ејҖе§Ӣж”»еҮ»
p1, q1 = Williams_p_plus_1_attack(n)
if p1 and q1:
    print("williams's p+1 factorization successful!")
    print(f"p = {p1}")
    print(f"q = {q1}")

# pollard's p-1 factorization
def pollard_p_minus_1(n, B=10**5): # Bдёәзҙ ж•°иЎЁзҡ„дёҠйҷҗ,еҸҜд»Ҙж №жҚ®е®һйҷ…еҫҖдёҠи°ғ
    a = 2  # йҖҡеёёйҖү2дҪңдёәеҹә
    for p in primerange(2, B):
        e = int(isqrt(n).bit_length() / p.bit_length())
        a = pow(a, pow(p, e), n)
    g = gcd(a - 1, n)
    if 1 < g < n:
        return g, n // g
    else:
        return None

p2, q2 = pollard_p_minus_1(n)
